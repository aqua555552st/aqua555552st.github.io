Spring模拟—Spring基于注解配置的程序

  1. 自己写一个简单的Spring容器，通过读取类的注解(@Component @Controller @Service @Reponsitory)，将对象注入到IOC容器中。
  2. 也就是说。不使用Spring原生框架，自己使用IO+Annotation+反射+集合 实现，打通Spring注解方式的计数痛点。

项目结构：

MyComponent类

package com.lhyedu.spring.component;

import org.springframework.stereotype.Component;

/**
 * @Component:标识该类是个组件，是一个通用的注解
 */
@Component("lhy1")
public class MyComponent {

}

ComponentScan注解

==自定义注解==

package com.lhyedu.spring.annotation;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

//@Target(ElementType.TYPE)指定我们的ComponentScan注解可以修饰Type程序元素
@Target(ElementType.TYPE)
//@Retention(RetentionPolicy.RUNTIME)指定注解保留范围
@Retention(RetentionPolicy.RUNTIME)

public @interface ComponentScan {
    String value() default "";
}

LhySpringConfig类

作用：类似于Spring原生框架下的 ==Beans.xml文件==

package com.lhyedu.spring.annotation;

/**
 * 这是一个配置之类
 * 作用：类似：原生Spring框架的 beans.xml文件
 */

@ComponentScan(value = "com.lhyedu.spring.component")
public class LhySpringConfig {
}

LhySpringApplicationContext类

作用：类似于Spring原生框架下的 ==ioc容器==

package com.lhyedu.spring.annotation;

import org.springframework.stereotype.Component;
import org.springframework.stereotype.Controller;
import org.springframework.stereotype.Repository;
import org.springframework.stereotype.Service;

import java.io.File;
import java.net.URL;
import java.util.concurrent.ConcurrentHashMap;

/**
 * 类似于Spring原生框架的 IOC
 */
public class LhySpringApplicationContext {
    private Class configClass;

    //ioc存放的就是通过反射创建的对象
    private final ConcurrentHashMap<String, Object> ioc = new ConcurrentHashMap<>();

    //构造器
    public LhySpringApplicationContext(Class configClass) {
        this.configClass = configClass;
        System.out.println("this.configClass = " + this.configClass);
        //获取要扫描的包
        //1.得到configClass的 @ComponentScan(value = "com.lhyedu.spring.component")注解
        ComponentScan componentScan =
                (ComponentScan) this.configClass.getDeclaredAnnotation(ComponentScan.class);

        //2.通过ComponentScan的value得到要扫描的包
        String path = componentScan.value();

        //3.获取扫描路径下的所有类文件(.class文件)
        //(1)先要得到类的加载器
        ClassLoader classLoader = LhySpringApplicationContext.class.getClassLoader();

        //(2)通过类加载器 获取到要扫描包的url
        //将"com.lhyedu.spring.component"换为"com/lhyedu/spring/component"
        path = path.replace(".", "/");
        System.out.println(path);
        //file:/D:/lhy_spring/spring5/out/production/spring5/com/lhyedu/spring/component
        URL resource = classLoader.getResource(path);

        //3.将要加载的资源路径下的文件进行遍历(I/O)
        File file = new File(resource.getFile());
        //文件可能是:普通文件/目录文件
        if (file.isDirectory()) {//若为目录文件

            File[] files = file.listFiles();
            for (File f : files) {//获取当前目录的所有文件(.class文件)
                String fileAbsolutePath = f.getAbsolutePath();
               // System.out.println("======================================");
                //System.out.println("文件的绝对路径=" + fileAbsolutePath);

                //(1)获取类名
                //file:/D:/.../com/lhyedu/spring/component/UserDao.class
                //切割成:UserDao
                String className = fileAbsolutePath
                        .substring(fileAbsolutePath.lastIndexOf("\\") + 1,
                                fileAbsolutePath.indexOf(".class"));
                //className = UserDao

                //(2)获取类的全路径(全类名)
                String classFullPath =
                        path.replace("/", ".") + "." + className;

                //(3)判断该类是否需要注入到容器中(过滤没加注解的类)
                try {
                    //这是就得到了该类(classFullPath)对应的class对象
                    Class<?> aClass = classLoader.loadClass(classFullPath);
                    //判断是否有注解
                    if (aClass.isAnnotationPresent(Component.class) ||
                        aClass.isAnnotationPresent(Controller.class) ||
                        aClass.isAnnotationPresent(Service.class) ||
                        aClass.isAnnotationPresent(Repository.class)) {

                        //将注解的value设为对象的名字
                        //仅以Component为例，其他同理...
                        if (aClass.isAnnotationPresent(Component.class)) {
                            Component component = aClass.getDeclaredAnnotation(Component.class);
                            String id = component.value();
                            if (!"".equals(id)) {
                                className = id;
                            }
                        }

                        //这是就可以反射对象，并放入容器中
                        Class<?> clazz = Class.forName(classFullPath);
                        Object instance = clazz.newInstance();
                        //放入到容器中
                        ioc.put(className, instance);
                    }
                } catch (Exception e) {
                    e.printStackTrace();
                }

            }
        }
    }

    public Object getBean(String name) {
        return ioc.get(name);
    }
}

Test类(主方法)

package com.lhyedu.spring.annotation;

public class Test {
    public static void main(String[] args) {
        //这里相当于：
        //ApplicationContext ioc =
        //    new ClassPathXmlApplicationContext("beans03.xml");
        LhySpringApplicationContext ioc =
                new LhySpringApplicationContext(LhySpringConfig.class);

        Object bean = ioc.getBean("lhy1");
        System.out.println(bean);
    }
}
