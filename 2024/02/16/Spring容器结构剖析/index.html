Spring容器结构剖析

[TOC]

package com.lhyedu.spring.test;

import com.lhyedu.spring.bean.Monster;
import org.junit.jupiter.api.Test;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class SpringBeanTest {
    @Test
    public void testFirst() {
        ApplicationContext ioc =
                new ClassPathXmlApplicationContext("beans.xml");

        Monster monster01 = ioc.getBean("monster01", Monster.class);
        System.out.println(monster01);//断点
    }
}

beanDefinitionMap

该类的所有属性

ioc -> beanfactory

beanfactory -> beanDefinitionMap

注意：beanDefinitionMap的数据类型是：ConcurrentHashMap

beanDefinitionMap的 属性

①table[]数组

保存配置的bean信息。

table[]数组的数据类型是：ConcurrentHashMap的内部类：ConcurrentHashMap$Node

②propertyValues

保存配置bean的”属性”信息。

singletonObjects

注意：singletonObjects的数据类型也是：ConcurrentHashMap

也维护一个table[]属性，保存创建出来的对象。

总结

xml不指定id

在beans.xml文件中，注入两个Monster对象，但均不指定id：

<bean class="com.lhyedu.spring.bean.Monster">
    <property name="monsterId" value="100"/>
    <property name="name" value="牛魔王"/>
    <property name="skill" value="牛魔王拳"/>
</bean>

<bean class="com.lhyedu.spring.bean.Monster">
    <property name="monsterId" value="200"/>
    <property name="name" value="白骨精"/>
    <property name="skill" value="吃人"/>
</bean>

运行会不会报错？

答：不会。系统会默认分配id，分配规则是全类名#1，全类名#2，全类名#n。

package com.lhyedu.spring.test;

import org.junit.jupiter.api.Test;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class SpringBeanTest {
    @Test
    public void testFirst() {
        //1.创建IOC容器对象
        ApplicationContext ioc =
                new ClassPathXmlApplicationContext("beans.xml");

        //BeanDefinitionNames[]:保存对象的变量名，便于查找
        String[] beanDefinitionNames = ioc.getBeanDefinitionNames();
        for (String s : beanDefinitionNames) {
            System.out.println(s);
        }
    }
}
