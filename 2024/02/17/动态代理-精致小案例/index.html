动态代理方式

注意：理解动态代理，是掌握AOP的核心

动态代理解决思路，在调用方法时，使用反射机制，根据方法去决定调用哪个对象。

VehicleProxyProvider类

该类可以返回一个代理对象

package com.lhyedu.spring.aop.proxy2;
import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

//该类可以返回一个 代理对象
public class VehicleProxyProvider {
    //定义一个属性
    //设置一个将来真正要运行的对象，只要是实现了Vehicle接口的就ok
    private Vehicle target_vehicle;

    //构造器
    public VehicleProxyProvider(Vehicle target_vehicle) {
        this.target_vehicle = target_vehicle;
    }

    //编写一个方法，返回一个代理对象(重点！！！)
    public Vehicle getProxy() {
        /**
         *     public static Object newProxyInstance(ClassLoader loader,
         *                                           Class<?>[] interfaces,
         *                                           InvocationHandler h)
         * 解读：
         * 0.Proxy.newProxyInstance()：可以返回一个"代理对象"
         * 1.ClassLoader loader：类的加载器
         * 2.Class<?>[] interfaces：将来要代理的对象的接口信息
         * 3.InvocationHandler h：调用处理器/对象 有一个非常重要的方法invoke
         */
        //1.得到类加载器
        ClassLoader cl = target_vehicle.getClass().getClassLoader();

        //2.得到 被代理/被执行对象 的接口信息
        Class<?>[] interfaces = target_vehicle.getClass().getInterfaces();

        //3.创建InvocationHandler 对象
        /**
         * public interface InvocationHandler {
         *     public Object invoke(Object proxy, Method method, Object[] args)
         *     throws Throwable;//接口的抽象方法
         * }
         */
        //因为InvocationHandler是接口 -> 可以通过匿名对象方式来创建该方式
        InvocationHandler invocationHandler = new InvocationHandler() {
            @Override
            //invoke():将来执行target_vehicle的方法时，会调用到的
            //Object o：代理对象
            //Method method：反射中的方法类
            //Object[] args：表示代理对象的方法的参数
            //返回：代理对象调用(invoke)后返回的结果
            public Object invoke(Object o, Method method, Object[] args)
                    throws Throwable {
                System.out.println("交通工具开始运行了..");
                //执行(重点！)。以下是我们的反射基础
                Object result = method.invoke(target_vehicle, args);

                System.out.println("交通工具停止运行了..");
                return result;
            }
        };

        //获取代理对象，该对象可以代理执行方法
        Vehicle proxy = (Vehicle) Proxy.newProxyInstance(cl, interfaces, invocationHandler);
        return proxy;
    }
}
@Test
public void proxySolution() {
    //创建一个Ship对象
    Vehicle vehicle = new Ship();
    //创建VehicleProxyProvider对象，并传入要代理的对象
    VehicleProxyProvider vehicleProxyProvider = new VehicleProxyProvider(vehicle);

    //获取代理对象(非常关键!!!) -> 代理执行方法
    //1.proxy编译类型：Vehicle
    //2.proxy运行类型：class com.sun.proxy.$Proxy9(代理类型)
    Vehicle proxy = vehicleProxyProvider.getProxy();
    System.out.println("(proxy运行类型是:" + proxy.getClass() + ")");
    proxy.run();
}

注意：proxy的编译类型&运行类型

执行proxy.run()语句时：

  • 没有执行Ship对象的run()方法
  • 通过：反射 + 动态代理机制
  • 执行的是proxy对象的newProxyInstance方法的参数(即：InvocationHandler实现类对象)
(proxy运行类型是:class com.sun.proxy.$Proxy9)

课后作业—动态代理深入[横切关注点]

有一个SmartAnimal接口，可以完成简单的加减法，要求在执行getSum()和getSub()时，输出执行前，执行过程，执行后的日志输出，思考如何用动态代理方式实现？

SmartAnimalable接口

package com.lhyedu.spring.aop.proxy;

public interface SmartAnimalable {
    float getSum(float i, float j);
    float getSub(float i, float j);
}

SmartDog类

package com.lhyedu.spring.aop.proxy;

public class SmartDog implements SmartAnimalable {
    @Override
    public float getSum(float i, float j) {
        return i + j;
    }

    @Override
    public float getSub(float i, float j) {
        return i - j;
    }
}

MyProxyProvide类[重点]

package com.lhyedu.spring.aop.proxy;
import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;
import java.util.Arrays;

public class MyProxyProvide {
    private SmartAnimalable target_obj;
    //构造器
    public MyProxyProvide(SmartAnimalable target_obj) {
        this.target_obj = target_obj;
    }

    public SmartAnimalable getProxy() {
        //1.得到类加载器
        ClassLoader loader = target_obj.getClass().getClassLoader();

        //2.得到接口信息
        Class<?>[] interfaces = target_obj.getClass().getInterfaces();

        //2.得到InvocationHandler实现类
        InvocationHandler h = new InvocationHandler() {
            @Override
            public Object invoke(Object proxy, Method method, Object[] args)
                    throws Throwable {
                Object result = null;
                try {
                    //这里从AOP看，就是一个横切关注点-"前置通知"
                    System.out.println("日志--方法名:" + method.getName()
                            + "--方法参数:" + Arrays.asList(args));

                    result = method.invoke(target_obj, args);

                    //也是一个横切关注点-"返回通知"
                    System.out.println("日志--方法名:" + method.getName()
                            + "--方法正常结束--返回结果:" + result);
                } catch (Exception e) {
                    e.printStackTrace();
                    //反射执行方法出现异常-> 横切关注点-"异常通知"
                    System.out.println("方法执行异常-日志-方法名:" +
                                       method.getName() + "--异常类型:"
                                       + e.getClass().getName());
                } finally {
                    //横切关注点-"最终通知"
                    System.out.println("方法最终结束-日志-方法名-" +
                                       method.getName());
                }
                return result;
            }
        };

        SmartAnimalable s = (SmartAnimalable) Proxy.newProxyInstance(loader, interfaces, h);
        return s;
    }
}
@Test
public void proxySolution2() {
    SmartAnimalable smartDog = new SmartDog();

    MyProxyProvide myProxyProvide = new MyProxyProvide(smartDog);

    SmartAnimalable proxy = myProxyProvide.getProxy();
    proxy.getSum(10.87f, 89.7f);
    System.out.println("========================================");
    proxy.getSub(10.87f, 89.7f);
}

问题

在MyProxyProvide.java中，我们输出的语句功能比较弱，而且是硬编码。在实际开发中，我们希望是以一个方法，嵌入到真正执行的目标方法前，怎么办？

改进：将以上几种"横切关注点"打包成函数，即可以在这些地方写日志和其他业务逻辑

土方法解决

将横切关注点-"前置通知"、横切关注点-"返回通知"等封装到AOP类中。这里LHYAOP视为一个切面类，达成解耦效果。

LHYAOP类[简易的AOP]

package com.lhyedu.spring.aop.proxy3;

import java.lang.reflect.Method;
import java.util.Arrays;

//极简的AOP类
public class LHYAOP {
    //函数参数 = invoke()的参数
    //将 横切关注点-"前置通知" 封装成before方法
    public static void before(Object proxy, Method method, Object[] args) {
        System.out.println("LHYAOP-before-日志--方法名:" + method.getName()
                + "--方法参数:" + Arrays.asList(args));
    }

    //将 横切关注点-"返回通知" 封装成after方法
    public static void after(Method method, Object result) {
        System.out.println("LHYAOP-after-日志--方法名:" + method.getName()
                + "--方法正常结束--返回结果:" + result);
    }
}

MyProxyProvide类

package com.lhyedu.spring.aop.proxy3;
import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

public class MyProxyProvide {
    private SmartAnimalable target_obj;

    //构造器
    public MyProxyProvide(SmartAnimalable target_obj) {
        this.target_obj = target_obj;
    }


    public SmartAnimalable getProxy() {

        //1.得到类加载器
        ClassLoader loader = target_obj.getClass().getClassLoader();

        //2.得到接口信息
        Class<?>[] interfaces = target_obj.getClass().getInterfaces();

        //2.得到InvocationHandler实现类
        InvocationHandler h = new InvocationHandler() {
            @Override
            public Object invoke(Object proxy, Method method, Object[] args)
                    throws Throwable {
                Object result = null;
                try {
                    //横切关注点-"前置通知"
                    LHYAOP.before(target_obj, method, args);

                    result = method.invoke(target_obj, args);

                    //横切关注点-"返回通知"
                    LHYAOP.after(method, result);
                } catch (Exception e) {
                    e.printStackTrace();
                    //反射执行方法出现异常-> 横切关注点-"异常通知"
                    System.out.println("方法执行异常-日志-方法名:" 
                                       + method.getName()
                            + "--异常类型:" + e.getClass().getName());
                } finally {
                    //横切关注点-"最终通知"
                    System.out.println("方法最终结束-日志-方法名-" 
                                       + method.getName());
                }
                return result;
            }
        };

        SmartAnimalable s = (SmartAnimalable) Proxy.newProxyInstance(loader, interfaces, h);
        return s;
    }
}
@Test
public void proxySolution2() {
    SmartAnimalable smartDog = new SmartDog();

    SmartAnimalable proxy = myProxyProvide.getProxy();
    proxy.getSum(10.87f, 89.7f);
    System.out.println("========================================");
    proxy.getSub(10.87f, 89.7f);

}
